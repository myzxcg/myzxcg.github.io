иҝҷдёӘзү№еҫҒи§ЈеҶіж–№жі•жңүеҫҲеӨҡпјҢжҜ”еҰӮпјҡ

иҝҳжңүдёҖз§Қж–№жі•е°ұжҳҜжҺҘдёӢжқҘд»Ӣз»Қзҡ„пјҡи®ҫзҪ®iptablesпјҢеҸӘе…Ғи®ёlocalhostи®ҝй—®cs listenerзӣ‘еҗ¬зҡ„з«ҜеҸЈгҖӮе°ҶеӨ–йғЁиҜ·жұӮйҖҡиҝҮnginxиҪ¬еҸ‘еҲ°localhostдёҠзҡ„listenerгҖӮиҝҳеҸҜд»ҘеңЁnginxдёҠи®ҫзҪ®иҝҮж»Өи§„еҲҷжқҘе…Ғи®ёзү№е®ҡиҜ·жұӮгҖӮ

cloudflareзј“еӯҳ

cloudflareиғҪејҖеҗҜејҖеҸ‘жЁЎејҸпјҢжқҘзҰҒз”Ёзј“еӯҳпјҢдҪҶжҳҜеҸӘжңү3дёӘе°Ҹж—¶гҖӮеҸҜд»ҘйҖҡиҝҮйЎөйқўи§„еҲҷжқҘж°ёд№…и®ҫзҪ®зј“еӯҳи§„еҲҷгҖӮ

еӣ дёәcsй…ҚзҪ®ж–Үд»¶дёӯи®ҫзҪ®зҡ„uriйғҪжҳҜjsз»“е°ҫзҡ„пјҢжүҖд»ҘиҝҷйҮҢдҪҝз”Ё*jsжқҘеҢ№й…ҚжүҖжңүuriгҖӮ

еқ‘зӮ№

й…ҚзҪ®е®ҢcdnеҗҺпјҢдјҡеҸ‘зҺ°жңәеҷЁиғҪдёҠзәҝпјҢдҪҶжҳҜжү§иЎҢе‘Ҫд»Өж— еӣһжҳҫгҖӮиҝҷдёӘй—®йўҳеӣ°жү°дәҶжҲ‘еҘҪеҚҠеӨ©пјҢдёҖзӣҙд»ҘдёәжҳҜcloudlflareзј“еӯҳй…ҚзҪ®й—®йўҳпјҢз„¶иҖҢзҰҒз”Ёзј“еӯҳеҗҺй—®йўҳдҫқж—§еӯҳеңЁгҖӮжңҖеҗҺйҖҗжӯҘе®ҡдҪҚеҲ°жҳҜcsй…ҚзҪ®ж–Үд»¶зҡ„й—®йўҳгҖӮ

жҲ‘з”Ёзҡ„жҳҜjquery-c2.4.0.profileй…ҚзҪ®ж–Үд»¶пјҢе®ғи®ҫзҪ®header "Content-Type" "application/javascript; charset=utf-8";дёәе“Қеә”еӨҙгҖӮдҝ®ж”№жүҖжңүе“Қеә”еӨҙдёәпјҡheader "Content-Type" "application/*; charset=utf-8";еҚіеҸҜжӯЈеёёжү§иЎҢе‘Ҫд»ӨеӣһжҳҫгҖӮ

иҝҷйҮҢзҡ„mime-typeеҰӮжһңдёәapplication/javascriptгҖҒtext/htmlзӯүпјҢжңәеҷЁжү§иЎҢе‘Ҫд»Өе°ұж— жі•еӣһжҳҫгҖӮжҲ‘жҖҖз–‘жҳҜcdnдјҡжЈҖжөӢе“Қеә”еӨҙcontent-typeзҡ„еҖјпјҢеҰӮжһңжҳҜдёҖдәӣйқҷжҖҒж–Үд»¶зҡ„mime-typeеҸҜиғҪе°ұеҜјиҮҙиҝҷдёӘй—®йўҳгҖӮ

иҮіжӯӨпјҢcdnе°ұй…ҚзҪ®е®ҢдәҶгҖӮе№¶дё”иғҪйҖҡиҝҮcdnжӯЈеёёдёҠзәҝгҖӮ

еҰӮжһңиғҪжүҫеҲ°иғҪиҝӣиЎҢеҹҹеүҚзҪ®жҠҖжңҜзҡ„cdnпјҢжҺЁиҚҗиҝҳжҳҜдҪҝз”ЁеҹҹеүҚзҪ®гҖӮcloudfalre workerеҸҜд»ҘиҜҙеҸӘжҳҜдёҖдёӘжӣҝд»Је“ҒгҖӮ

cloudflare workerиғҪеӨҹжү§иЎҢж— жңҚеҠЎеҷЁеҮҪж•°пјҢе…Қиҙ№з”ЁжҲ·жңү10дёҮиҜ·жұӮ/жҜҸеӨ©зҡ„йўқеәҰгҖӮе№¶дё”дҪ иғҪиҮӘе®ҡд№үworkers.devзҡ„еӯҗеҹҹгҖӮжҲ‘д»¬еҸҜд»Ҙзј–еҶҷjsеӨ„зҗҶд»ҘеҸҠиҪ¬еҸ‘иҜ·жұӮгҖӮ

jsи„ҡжң¬еҰӮдёӢпјҢйңҖиҰҒи®ҫзҪ®X-Forwarded-ForеӨҙпјҢдёҚз„¶дёҠзәҝзҡ„ipжҳҜworkerзҡ„ipv6ең°еқҖгҖӮ

let upstream = 'https://img.xxx.com'

addEventListener('fetch', event => {
    event.respondWith(fetchAndApply(event.request));
})

async function fetchAndApply(request) {
    const ipAddress = request.headers.get('cf-connecting-ip') || '';
    let requestURL = new URL(request.url);
    let upstreamURL = new URL(upstream);
    requestURL.protocol = upstreamURL.protocol;
    requestURL.host = upstreamURL.host;
    requestURL.pathname = upstreamURL.pathname + requestURL.pathname;

    let new_request_headers = new Headers(request.headers);
    new_request_headers.set("X-Forwarded-For", ipAddress);
    let fetchedResponse = await fetch(
        new Request(requestURL, {
            method: request.method,
            headers: new_request_headers,
            body: request.body
        })
    );
    let modifiedResponseHeaders = new Headers(fetchedResponse.headers);
    modifiedResponseHeaders.delete('set-cookie');
    return new Response(
        fetchedResponse.body,
        {
            headers: modifiedResponseHeaders,
            status: fetchedResponse.status,
            statusText: fetchedResponse.statusText
        }
    );
}
